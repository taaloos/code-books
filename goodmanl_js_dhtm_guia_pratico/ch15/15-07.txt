function dayPart() { 
    var oneDate = new Date(); 
    var theHour = oneDate.getHours(); 
    if (theHour < 12) { 
        return "morning"; 
    } else if (theHour < 18) { 
        return "afternoon"; 
    } else { 
        return "evening"; 
    } 
}
